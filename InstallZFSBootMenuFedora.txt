REM Fedora Server installer -> switch to tty2 shell -> sync clock/RTC -> bootstrap ZFSBootMenu installer from feature/fedora-support
REM Assumes the Fedora Server installer has already booted and wired networking is available.
DEFAULT_DELAY 200
DELAY 12000

CTRL ALT F2
DELAY 2500

STRING nmcli networking on > /dev/null 2>&1 || true
ENTER
DELAY 300

STRING timedatectl set-ntp true > /dev/null 2>&1 || true
ENTER
DELAY 300

STRING hwclock --systohc 2>/dev/null || true
ENTER
DELAY 300

STRING pkg_manager="$(command -v dnf || command -v dnf5 || command -v microdnf)"
ENTER
DELAY 300

STRING [ -n "$pkg_manager" ] || echo "No Fedora package manager found; assuming curl is already installed"
ENTER
DELAY 300

STRING command -v curl >/dev/null 2>&1 || { [ -n "$pkg_manager" ] && "$pkg_manager" install -y curl; }
ENTER
DELAY 300

STRING curl -fsSLo setup-zfsbootmenu.sh https://raw.githubusercontent.com/MuffinSmith/zfsbootmenu-autoinstaller/feature/fedora-support/setup-zfsbootmenu.sh
ENTER
DELAY 300

STRING chmod +x setup-zfsbootmenu.sh
ENTER
DELAY 300

STRING ./setup-zfsbootmenu.sh
ENTER
